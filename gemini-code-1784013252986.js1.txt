// --- FUNGSI UTAMA UNTUK MENGHUBUNGI API GEMINI ---
const generateAIContent = async (prompt, systemInstruction = "", userApiKey = "") => {
  if (!userApiKey) {
    throw new Error("API_KEY_MISSING");
  }

  // Model gratis & stabil yang didukung Google AI Studio
  const modelName = "gemini-1.5-flash"; 
  const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/${modelName}:generateContent?key=${userApiKey}`;

  const payload = {
    contents: [{ parts: [{ text: prompt }] }],
  };

  if (systemInstruction) {
    payload.systemInstruction = { parts: [{ text: systemInstruction }] };
  }

  try {
    const response = await fetch(apiUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });

    if (response.status === 400 || response.status === 403) {
      throw new Error("API_KEY_INVALID");
    }

    if (!response.ok) {
      throw new Error(`API error ${response.status}`);
    }

    const result = await response.json();
    if (result.candidates && result.candidates.length > 0) {
      return result.candidates[0].content.parts[0].text;
    }
    throw new Error("Gagal memproses jawaban dari AI");
  } catch (error) {
    throw error;
  }
};

// --- KOMPONEN UTAMA APLIKASI ---
export default function CreatorCopilotApp() {
  const [apiKey, setApiKey] = useState("");
  const [isSaved, setIsSaved] = useState(false);
  const [activeTab, setActiveTab] = useState("planner");
  const [inputText, setInputText] = useState("");
  const [aiOutput, setAiOutput] = useState("");
  const [loading, setLoading] = useState(false);
  const [errorMsg, setErrorMsg] = useState("");

  // Ambil API Key dari memori browser saat aplikasi pertama kali dibuka
  useEffect(() => {
    const savedKey = localStorage.getItem("user_gemini_api_key");
    if (savedKey) {
      setApiKey(savedKey);
      setIsSaved(true);
    }
  }, []);

  // Simpan API Key ke memori browser
  const handleSaveApiKey = (e) => {
    e.preventDefault();
    if (apiKey.trim()) {
      localStorage.setItem("user_gemini_api_key", apiKey.trim());
      setIsSaved(true);
      setErrorMsg("");
      alert("API Key berhasil disimpan!");
    } else {
      alert("Silakan masukkan API Key yang valid terlebih dahulu.");
    }
  };

  // Hapus API Key dari memori browser
  const handleResetApiKey = () => {
    localStorage.removeItem("user_gemini_api_key");
    setApiKey("");
    setIsSaved(false);
    alert("API Key telah dihapus.");
  };

  // Fungsi saat Pengguna menekan tombol "Generate"
  const handleGenerate = async () => {
    if (!apiKey) {
      setErrorMsg("Silakan masukkan dan simpan API Key Anda terlebih dahulu di kolom atas!");
      return;
    }

    setLoading(true);
    setErrorMsg("");
    setAiOutput("");

    let systemPrompt = "Anda adalah asisten pembuat konten (Creator Copilot) yang kreatif dan profesional.";
    let userPrompt = inputText;

    if (activeTab === "planner") {
      systemPrompt += " Tugas Anda adalah membuat ide konten bulanan terstruktur.";
      userPrompt = `Buatkan jadwal ide konten 30 hari berdasarkan topik berikut: ${inputText}`;
    } else if (activeTab === "script") {
      systemPrompt += " Tugas Anda adalah menulis naskah video/audio yang menarik perhatian penonton.";
      userPrompt = `Buatkan naskah video TikTok/Reels berdurasi 60 detik tentang: ${inputText}`;
    }

    try {
      const result = await generateAIContent(userPrompt, systemPrompt, apiKey);
      setAiOutput(result);
    } catch (err) {
      if (err.message === "API_KEY_INVALID") {
        setErrorMsg("API Key salah atau tidak valid. Silakan periksa kembali di Google AI Studio.");
      } else if (err.message === "API_KEY_MISSING") {
        setErrorMsg("API Key belum diisi.");
      } else {
        setErrorMsg("Terjadi gangguan jaringan atau kegagalan sistem AI.");
      }
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-800 font-sans p-4 md:p-8">
      <div className="max-w-4xl mx-auto space-y-6">
        
        {/* HEADER */}
        <header className="flex items-center space-x-3 pb-6 border-b border-slate-200">
          <div className="p-2 bg-indigo-600 text-white rounded-xl">
            <Icons.Brain />
          </div>
          <div>
            <h1 className="text-2xl font-bold tracking-tight text-slate-900">Creator Copilot Lite</h1>
            <p className="text-sm text-slate-500">Asisten pembuat konten otomatis bertenaga AI</p>
          </div>
        </header>

        {/* 1. KOTAK PENGATURAN API KEY (PENTING UNTUK PEMBELI) */}
        <section className="bg-white p-5 rounded-2xl shadow-sm border border-slate-200">
          <div className="flex items-center space-x-2 text-indigo-600 font-semibold mb-3">
            <Icons.Key />
            <h2>Pengaturan API Key Gemini Anda</h2>
          </div>
          <p className="text-xs text-slate-500 mb-4 leading-relaxed">
            Aplikasi ini berjalan mandiri di browser Anda. Masukkan <strong>Gemini API Key</strong> gratis Anda dari Google AI Studio agar aplikasi bisa mulai bekerja. Kunci ini hanya disimpan di browser Anda pribadi.
          </p>

          <form onSubmit={handleSaveApiKey} className="flex flex-col sm:flex-row gap-3">
            <input
              type="password"
              placeholder={isSaved ? "••••••••••••••••••••••••••••••••" : "Masukkan API Key Gemini Anda di sini..."}
              value={apiKey}
              onChange={(e) => {
                setApiKey(e.target.value);
                if (isSaved) setIsSaved(false);
              }}
              disabled={isSaved}
              className="flex-1 px-4 py-2.5 rounded-xl border border-slate-300 text-sm focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 disabled:bg-slate-100 disabled:text-slate-400"
            />
            {isSaved ? (
              <button
                type="button"
                onClick={handleResetApiKey}
                className="px-5 py-2.5 bg-rose-50 text-rose-600 text-sm font-medium rounded-xl hover:bg-rose-100 transition-colors"
              >
                Ganti Key
              </button>
            ) : (
              <button
                type="submit"
                className="px-5 py-2.5 bg-indigo-600 text-white text-sm font-medium rounded-xl hover:bg-indigo-700 transition-colors"
              >
                Simpan Key
              </button>
            )}
          </form>
        </section>

        {/* 2. AREA NAVIGASI FITUR */}
        <div className="flex space-x-2 p-1 bg-slate-200/60 rounded-xl max-w-xs">
          <button
            onClick={() => { setActiveTab("planner"); setAiOutput(""); }}
            className={`flex-1 flex items-center justify-center space-x-2 py-2 px-3 rounded-lg text-xs font-semibold transition-all ${
              activeTab === "planner" ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500 hover:text-slate-800'
            }`}
          >
            <Icons.Calendar />
            <span>Ide Konten 30 Hari</span>
          </button>
          <button
            onClick={() => { setActiveTab("script"); setAiOutput(""); }}
            className={`flex-1 flex items-center justify-center space-x-2 py-2 px-3 rounded-lg text-xs font-semibold transition-all ${
              activeTab === "script" ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500 hover:text-slate-800'
            }`}
          >
            <Icons.PenTool />
            <span>Naskah Video</span>
          </button>
        </div>

        {/* 3. INPUT FORM INPUTAN USER */}
        <main className="grid grid-cols-1 md:grid-cols-2 gap-6">
          <div className="bg-white p-6 rounded-2xl border border-slate-200 shadow-sm flex flex-col justify-between">
            <div className="space-y-4">
              <label className="block text-sm font-semibold text-slate-700">
                {activeTab === "planner" ? "Topik Utama atau Niche Anda" : "Topik / Konsep Naskah Video"}
              </label>
              <textarea
                rows={5}
                placeholder={activeTab === "planner" ? "Contoh: Tips diet keto untuk pemula, atau Tutorial memasak praktis anak kos..." : "Contoh: Cara menghemat uang belanja bulanan, atau 3 kesalahan saat mencuci wajah..."}
                value={inputText}
                onChange={(e) => setInputText(e.target.value)}
                className="w-full px-4 py-3 border border-slate-300 rounded-xl text-sm focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500"
              />
            </div>

            {errorMsg && (
              <div className="mt-4 p-3 bg-red-50 text-red-700 text-xs rounded-xl border border-red-100">
                {errorMsg}
              </div>
            )}

            <button
              onClick={handleGenerate}
              disabled={loading || !inputText}
              className="mt-6 w-full flex items-center justify-center space-x-2 py-3 bg-slate-900 text-white rounded-xl font-medium hover:bg-slate-800 transition-colors disabled:bg-slate-200 disabled:text-slate-400"
            >
              {loading ? (
                <>
                  <Icons.Loader />
                  <span>Sedang meracik konten...</span>
                </>
              ) : (
                <span>Mulai Buat Sekarang ✨</span>
              )}
            </button>
          </div>

          {/* 4. TAMPILAN OUTPUT HASIL AI */}
          <div className="bg-white p-6 rounded-2xl border border-slate-200 shadow-sm flex flex-col min-h-[300px]">
            <h3 className="text-sm font-semibold text-slate-700 mb-4 border-b border-slate-100 pb-2">
              Hasil Rekomendasi AI
            </h3>
            
            {aiOutput ? (
              <div className="flex-1 whitespace-pre-line text-sm text-slate-600 overflow-y-auto max-h-[350px]">
                {aiOutput}
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center text-center text-slate-400">
                <Icons.Brain />
                <p className="mt-2 text-xs">Hasil AI Anda akan muncul di sini.</p>
              </div>
            )}
          </div>
        </main>

      </div>
    </div>
  );
}
